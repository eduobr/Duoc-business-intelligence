CREATE DATABASE EOH002
GO
USE EOH002

CREATE TABLE DMTIEMPO(
idTiempo INT NOT NULL PRIMARY KEY,
OrderDate datetime NOT NULL
);

CREATE TABLE DMPRODUCTO(
idProducto INT NOT NULL PRIMARY KEY,
Name nvarchar(50) NOT NULL,
ProductNumber nvarchar(25)NOT NULL
);

CREATE TABLE HTOTAL(
idHtotal INT NOT NULL PRIMARY KEY identity(1,1),
idProducto INT NOT NULL FOREIGN KEY REFERENCES DMPRODUCTO(idProducto),
idTiempo INT NOT NULL FOREIGN KEY REFERENCES DMTIEMPO(idTiempo),
cantidad INT NOT NULL
);
